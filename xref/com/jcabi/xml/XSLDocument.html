81       * Strips spaces of whitespace-only text nodes.
82       *
83       * <p>This will NOT remove
84       * existing indentation between Element nodes currently introduced by the
85       * constructor of {@link XMLDocument}. For example:
86       *
87       * <pre>
88       * {@code
89       * &lt;a&gt;
90       *           &lt;b> TXT &lt;/b>
91       *    &lt;/a>}
92       * </pre>
93       *
94       * becomes
95       *
96       * <pre>
97       * {@code
98       * &lt;a>
99       *     &lt;b> TXT &lt;/b>
100      * &lt;/a>}
101      * </pre>
102      *
103      * @since 0.14
104      */
105     public static final XSL STRIP = XSLDocument.make(
106         XSL.class.getResourceAsStream("strip.xsl")
107     );

383     @Override
384     public XML transform(final XML xml) {
385         final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
386         final DocumentBuilder builder;
387         try {
388             builder = factory.newDocumentBuilder();
389         } catch (final ParserConfigurationException ex) {
390             throw new IllegalArgumentException(
391                 String.format(
392                     "Failed to create new XML document by %s",
393                     factory.getClass().getName()
394                 ),
395                 ex
396             );
397         }
398         final Document target = builder.newDocument();
399         this.transformInto(xml, new DOMResult(target));
400         return new XMLDocument(target);
401     }

403     @Override
404     public String applyTo(final XML xml) {
405         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
406         this.transformInto(xml, new StreamResult(baos));
407         try {
408             return baos.toString(StandardCharsets.UTF_8.name());
409         } catch (final UnsupportedEncodingException ex) {
410             throw new IllegalArgumentException(
411                 "Failed to convert bytes into UTF-8 string",
412                 ex
413             );
414         }
415     }

417     /**
418      * Transform XML into result.
419      *
420      * We create {@link TransformerFactory} here on every transformation
421      * because {@link javax.xml.transform.URIResolver} must be set into
422      * it before making an instance of a transformer. Otherwise, it won't
423      * understand "xsl:import" statements.
424      *
425      * @param xml XML
426      * @param result Result
427      * @since 0.11
428      * @link <a href="https://stackoverflow.com/questions/4695489">Relevant SO question</a>
429      */
430     private void transformInto(final XML xml, final Result result) {
431         final Transformer trans = this.transformer();
432         final ConsoleErrorListener errors = new ConsoleErrorListener();
433         trans.setErrorListener(errors);
434         final long start = System.nanoTime();
435         try {
436             trans.transform(new DOMSource(xml.inner()), result);
437         } catch (final TransformerException ex) {
438             final StringBuilder summary = new StringBuilder(
439                 String.join("; ", errors.summary())
440             );
441             if (!summary.toString().equals(ex.getMessageAndLocation())) {
442                 summary.append("; ").append(ex.getMessageAndLocation());
443             }
444             throw new IllegalArgumentException(
445                 String.format(
446                     "Failed to transform by %s: %s",
447                     trans.getClass().getName(),
448                     summary
449                 ),
450                 ex
451             );
452         }
453         if (Logger.isTraceEnabled(this)) {
454             Logger.trace(
455                 this,
456                 "%s transformed XML in %[nano]s",
457                 trans.getClass().getName(),
458                 System.nanoTime() - start
459             );
460         }
461     }

463     /**
464      * Make a transformer.
465      * @return The transformer
466      */
467     private Transformer transformer() {
468         final TransformerFactory factory = TransformerFactory.newInstance();
469         factory.setURIResolver(this.sources);
470         final Transformer trans;
471         try {
472             trans = factory.newTransformer(
473                 new StreamSource(new StringReader(this.xsl), this.sid)
474             );
475         } catch (final TransformerConfigurationException ex) {
476             throw new IllegalArgumentException(
477                 String.format(
478                     "Failed to create transformer by %s",
479                     factory.getClass().getName()
480                 ),
481                 ex
482             );
483         }
484         for (final Map.Entry<String, Object> ent : this.params.entrySet()) {
485             trans.setParameter(ent.getKey(), ent.getValue());
486         }
487         return trans;
488     }

490     /**
491      * Prepare it for Saxon.
492      * @param trans The transformer
493      * @return The same
494      * @checkstyle ReturnCountCheck (5 lines)
495      */
496     @SuppressWarnings("deprecation")
497     private static Transformer forSaxon(final Transformer trans) {
498         final String type = trans.getClass().getCanonicalName();
499         if (!"net.sf.saxon.jaxp.TransformerImpl".equals(type)) {
500             return trans;
501         }
502         if (Version.getStructuredVersionNumber()[0] < 11) {
503             ((TransformerImpl) trans)
504                 .getUnderlyingController()
505                 .setMessageEmitter(new MessageWarner());
506         }
507         if (Version.getStructuredVersionNumber()[0] >= 11) {
508             ((TransformerImpl) trans)
509                 .getUnderlyingController()
510                 .setMessageHandler(
511                     message -> Logger.error(
512                         XSLDocument.class,
513                         "%s: %s",
514                         message.getLocation(),
515                         message.toString()
516                     )
517                 );
518         }
519         return trans;
520     }
