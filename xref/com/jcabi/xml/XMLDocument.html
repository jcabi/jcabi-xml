311     /**
312      * Private ctor.
313      * @param node The source
314      * @param ctx Namespace context
315      * @param lfe Is it a leaf node?
316      */
317     private XMLDocument(final Node node, final XPathContext ctx,
318         final boolean lfe) {
319         this.xml = XMLDocument.asString(node);
320         this.context = ctx;
321         this.leaf = lfe;
322         this.cache = node;
323     }

330     @Override
331     public Node node() {
332         final Node casted = Node.class.cast(this.cache);
333         final Node answer;
334         if (casted instanceof Document) {
335             answer = casted.cloneNode(true);
336         } else {
337             answer = XMLDocument.createImportedNode(casted);
338         }
339         return answer;
340     }

342     @Override
343     @SuppressWarnings
344         (
345             {
346                 "PMD.ExceptionAsFlowControl",
347                 "PMD.PreserveStackTrace"
348             }
349         )
350     public List<String> xpath(final String query) {
351         // @checkstyle FinalLocalVariableCheck (1 line)
352         List<String> items;
353         try {
354             final NodeList nodes = this.fetch(query, NodeList.class);
355             items = new ArrayList<>(nodes.getLength());
356             for (int idx = 0; idx < nodes.getLength(); ++idx) {
357                 final int type = (int) nodes.item(idx).getNodeType();
358                 if (type != (int) Node.TEXT_NODE
359                     && type != (int) Node.ATTRIBUTE_NODE
360                     && type != (int) Node.CDATA_SECTION_NODE) {
361                     throw new IllegalArgumentException(
362                         String.format(
363                             // @checkstyle LineLength (1 line)
364                             "Only text() nodes or attributes are retrievable with xpath() '%s': %d",
365                             query, type
366                         )
367                     );
368                 }
369                 items.add(nodes.item(idx).getNodeValue());
370             }
371         } catch (final XPathExpressionException ex) {
372             try {
373                 items = Collections.singletonList(
374                     this.fetch(query, String.class)
375                 );
376             } catch (final XPathExpressionException exp) {
377                 throw new IllegalArgumentException(
378                     // @checkstyle MultipleStringLiterals (1 line)
379                     String.format(
380                         "Invalid XPath query '%s' at %s: %s",
381                         query, XMLDocument.XFACTORY.getClass().getName(),
382                         ex.getLocalizedMessage()
383                     ),
384                     exp
385                 );
386             }
387         }
388         return new ListWrapper<>(items, Node.class.cast(this.cache), query);
389     }

400     @Override
401     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
402     public List<XML> nodes(final String query) {
403         final List<XML> items;
404         try {
405             final NodeList nodes = this.fetch(query, NodeList.class);
406             items = new ArrayList<>(nodes.getLength());
407             for (int idx = 0; idx < nodes.getLength(); ++idx) {
408                 items.add(
409                     new XMLDocument(
410                         nodes.item(idx),
411                         this.context, true
412                     )
413                 );
414             }
415         } catch (final XPathExpressionException ex) {
416             throw new IllegalArgumentException(
417                 String.format(
418                     "invalid XPath query '%s' by %s",
419                     query, XMLDocument.XFACTORY.getClass().getName()
420                 ), ex
421             );
422         }
423         return new ListWrapper<>(items, Node.class.cast(this.cache), query);
424     }

435     /**
436      * Clones a node and imports it in a new document.
437      * @param node A node to clone.
438      * @return A cloned node imported in a dedicated document.
439      */
440     private static Node createImportedNode(final Node node) {
441         final Document document;
442         try {
443             document = XMLDocument.DFACTORY.newDocumentBuilder().newDocument();
444         } catch (final ParserConfigurationException ex) {
445             throw new IllegalStateException(ex);
446         }
447         final Node imported = document.importNode(node, true);
448         document.appendChild(imported);
449         return imported;
450     }

452     /**
453      * Retrieve XPath query result. Supports returning {@link NodeList} and
454      * {@link String} types.
455      *
456      * <p>An {@link IllegalArgumentException} is thrown if the parameter
457      * passed is not a valid XPath expression or an unsupported type is
458      * specified.
459      *
460      * @param query XPath query
461      * @param type The return type
462      * @param <T> The type to return
463      * @return Result of XPath query
464      * @throws XPathExpressionException If an error occurs when evaluating XPath
465      */
466     @SuppressWarnings("unchecked")
467     private <T> T fetch(final String query, final Class<T> type)
468         throws XPathExpressionException {
469         final XPath xpath;
470         synchronized (XMLDocument.class) {
471             xpath = XMLDocument.XFACTORY.newXPath();
472         }
473         xpath.setNamespaceContext(this.context);
474         final QName qname;
475         if (type.equals(String.class)) {
476             qname = XPathConstants.STRING;
477         } else if (type.equals(NodeList.class)) {
478             qname = XPathConstants.NODESET;
479         } else {
480             throw new IllegalArgumentException(
481                 String.format(
482                     "Unsupported type: %s", type.getName()
483                 )
484             );
485         }
486         return (T) xpath.evaluate(query, Node.class.cast(this.cache), qname);
487     }

489     /**
490      * Transform node to String.
491      *
492      * @param node The DOM node.
493      * @return String representation
494      */
495     @SuppressWarnings("PMD.AvoidDuplicateLiterals")
496     private static String asString(final Node node) {
497         final StringWriter writer = new StringWriter();
498         try {
499             final Transformer trans;
500             synchronized (XMLDocument.class) {
501                 trans = XMLDocument.TFACTORY.newTransformer();
502             }
503             // @checkstyle MultipleStringLiterals (1 line)
504             trans.setOutputProperty(OutputKeys.INDENT, "yes");
505             trans.setOutputProperty(OutputKeys.VERSION, "1.0");
506             if (!(node instanceof Document)) {
507                 trans.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
508             }
509             synchronized (node) {
510                 trans.transform(
511                     new DOMSource(node),
512                     new StreamResult(writer)
513                 );
514             }
515         } catch (final TransformerConfigurationException ex) {
516             throw new IllegalStateException(ex);
517         } catch (final TransformerException ex) {
518             throw new IllegalArgumentException(ex);
519         }
520         return writer.toString();
521     }

523     /**
524      * Transform source to DOM node.
525      * @param source The source
526      * @return The node
527      */
528     private static Node transform(final Source source) {
529         final DOMResult result = new DOMResult();
530         try {
531             final Transformer trans;
532             synchronized (XMLDocument.class) {
533                 trans = XMLDocument.TFACTORY.newTransformer();
534             }
535             trans.transform(source, result);
536         } catch (final TransformerException ex) {
537             throw new IllegalStateException(ex);
538         }
539         return result.getNode();
540     }
