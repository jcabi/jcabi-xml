136     /**
137      * Public ctor, from XML as a text.
138      *
139      * <p>The object is created with a default implementation of
140      * {@link NamespaceContext}, which already defines a
141      * number of namespaces, for convenience, including:
142      *
143      * <pre> xhtml: http://www.w3.org/1999/xhtml
144      * xs: http://www.w3.org/2001/XMLSchema
145      * xsi: http://www.w3.org/2001/XMLSchema-instance
146      * xsl: http://www.w3.org/1999/XSL/Transform
147      * svg: http://www.w3.org/2000/svg</pre>;
148      *
149      * <p>In future versions we will add more namespaces (submit a ticket if
150      * you need more of them defined here).
151      *
152      * <p>An {@link IllegalArgumentException} is thrown if the parameter
153      * passed is not in XML format.
154      *
155      * @param text XML document body
156      */
157     public XMLDocument(@NotNull(message = "XML text can't be NULL")
158         final String text) {
159         this(
160             new DomParser(XMLDocument.DFACTORY, text).document(),
161             new XPathContext(),
162             false
163         );
164     }

274     /**
275      * Private ctor.
276      * @param node The source
277      * @param ctx Namespace context
278      * @param lfe Is it a leaf node?
279      */
280     private XMLDocument(final Node node, final XPathContext ctx,
281         final boolean lfe) {
282         this.xml = XMLDocument.asString(node);
283         this.context = ctx;
284         this.leaf = lfe;
285         this.cache = node;
286     }

288     @Override
289     public String toString() {

293     @Override
294     @NotNull(message = "node is never NULL")
295     public Node node() {
296         final Node castCache = Node.class.cast(this.cache);
297         final Node answer;
298         if (castCache instanceof Document) {
299             answer = castCache.cloneNode(true);
300         } else {
301             answer = this.createImportedNode(castCache);
302         }
303         return answer;
304     }

306     @Override
307     @NotNull(message = "list of texts is never NULL")
308     @SuppressWarnings
309         (
310             {
311                 "PMD.ExceptionAsFlowControl",
312                 "PMD.PreserveStackTrace"
313             }
314         )
315     public List<String> xpath(@NotNull final String query) {
316         List<String> items;
317         try {
318             final NodeList nodes = this.fetch(query, NodeList.class);
319             items = new ArrayList<String>(nodes.getLength());
320             for (int idx = 0; idx < nodes.getLength(); ++idx) {
321                 final int type = nodes.item(idx).getNodeType();
322                 if (type != Node.TEXT_NODE && type != Node.ATTRIBUTE_NODE
323                     && type != Node.CDATA_SECTION_NODE) {
324                     throw new IllegalArgumentException(
325                         String.format(
326                             // @checkstyle LineLength (1 line)
327                             "Only text() nodes or attributes are retrievable with xpath() '%s': %d",
328                             query, type
329                         )
330                     );
331                 }
332                 items.add(nodes.item(idx).getNodeValue());
333             }
334         } catch (final XPathExpressionException ex) {
335             try {
336                 items = Collections.singletonList(
337                     this.fetch(query, String.class)
338                 );
339             } catch (final XPathExpressionException exp) {
340                 throw new IllegalArgumentException(
341                     // @checkstyle MultipleStringLiterals (1 line)
342                     String.format(
343                         "invalid XPath query '%s' at %s",
344                         query, XMLDocument.XFACTORY.getClass().getName()
345                     ), exp
346                 );
347             }
348         }
349         return new ListWrapper<String>(items, this.node(), query);
350     }

361     @Override
362     @NotNull(message = "XML is never NULL")
363     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
364     public List<XML> nodes(@NotNull final String query) {
365         final List<XML> items;
366         try {
367             final NodeList nodes = this.fetch(query, NodeList.class);
368             items = new ArrayList<XML>(nodes.getLength());
369             for (int idx = 0; idx < nodes.getLength(); ++idx) {
370                 items.add(
371                     new XMLDocument(
372                         nodes.item(idx),
373                         this.context, true
374                     )
375                 );
376             }
377         } catch (final XPathExpressionException ex) {
378             throw new IllegalArgumentException(
379                 String.format(
380                     "invalid XPath query '%s' by %s",
381                     query, XMLDocument.XFACTORY.getClass().getName()
382                 ), ex
383             );
384         }
385         return new ListWrapper<XML>(items, this.node(), query);
386     }

395     /**
396      * Clones a node and imports it in a new document.
397      * @param node A node to clone.
398      * @return A cloned node imported in a dedicated document.
399      */
400     private Node createImportedNode(final Node node) {
401         final Document document;
402         try {
403             document = DFACTORY.newDocumentBuilder().newDocument();
404         } catch (final ParserConfigurationException ex) {
405             throw new IllegalStateException(ex);
406         }
407         final Node imported = document.importNode(node, true);
408         document.appendChild(imported);
409         return imported;
410     }

412     /**
413      * Retrieve XPath query result. Supports returning {@link NodeList} and
414      * {@link String} types.
415      *
416      * <p>An {@link IllegalArgumentException} is thrown if the parameter
417      * passed is not a valid XPath expression or an unsupported type is
418      * specified.
419      *
420      * @param <T> The type to return
421      * @param query XPath query
422      * @param type The return type
423      * @return Result of XPath query
424      * @throws XPathExpressionException If an error occurs when evaluating XPath
425      */
426     @SuppressWarnings("unchecked")
427     private <T> T fetch(final String query, final Class<T> type)
428         throws XPathExpressionException {
429         final XPath xpath;
430         synchronized (XMLDocument.class) {
431             xpath = XMLDocument.XFACTORY.newXPath();
432         }
433         xpath.setNamespaceContext(this.context);
434         final QName qname;
435         if (type.equals(String.class)) {
436             qname = XPathConstants.STRING;
437         } else if (type.equals(NodeList.class)) {
438             qname = XPathConstants.NODESET;
439         } else {
440             throw new IllegalArgumentException(
441                 String.format(
442                     "Unsupported type: %s", type.getName()
443                 )
444             );
445         }
446         return (T) xpath.evaluate(query, this.node(), qname);
447     }

449     /**
450      * Transform node to String.
451      *
452      * @param node The DOM node.
453      * @return String representation
454      */
455     private static String asString(final Node node) {
456         final StringWriter writer = new StringWriter();
457         try {
458             final Transformer trans;
459             synchronized (XMLDocument.class) {
460                 trans = XMLDocument.TFACTORY.newTransformer();
461             }
462             // @checkstyle MultipleStringLiterals (1 line)
463             trans.setOutputProperty(OutputKeys.INDENT, "yes");
464             trans.setOutputProperty(OutputKeys.VERSION, "1.0");
465             if (!(node instanceof Document)) {
466                 trans.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
467             }
468             synchronized (node) {
469                 trans.transform(
470                     new DOMSource(node),
471                     new StreamResult(writer)
472                 );
473             }
474         } catch (final TransformerConfigurationException ex) {
475             throw new IllegalStateException(ex);
476         } catch (final TransformerException ex) {
477             throw new IllegalArgumentException(ex);
478         }
479         return writer.toString();
480     }

482     /**
483      * Transform source to DOM node.
484      * @param source The source
485      * @return The node
486      */
487     private static Node transform(final Source source) {
488         final DOMResult result = new DOMResult();
489         try {
490             final Transformer trans;
491             synchronized (XMLDocument.class) {
492                 trans = XMLDocument.TFACTORY.newTransformer();
493             }
494             trans.transform(source, result);
495         } catch (final TransformerConfigurationException ex) {
496             throw new IllegalStateException(ex);
497         } catch (final TransformerException ex) {
498             throw new IllegalStateException(ex);
499         }
500         return result.getNode();
501     }
