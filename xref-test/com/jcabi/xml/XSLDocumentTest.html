50      @Test
51      public void makesXslTransformations() {
52          final XSL xsl = new XSLDocument(
53              StringUtils.join(
54                  "<xsl:stylesheet",
55                  " xmlns:xsl='http://www.w3.org/1999/XSL/Transform'",
56                  " version='2.0'>",
57                  "<xsl:template match='/'><done/>",
58                  "</xsl:template></xsl:stylesheet>"
59              )
60          );
61          MatcherAssert.assertThat(
62              xsl.transform(new XMLDocument("<a/>")),
63              XhtmlMatchers.hasXPath("/done")
64          );
65          MatcherAssert.assertThat(
66              xsl.transform(new XMLDocument("<a></a>")),
67              XhtmlMatchers.hasXPath("/done ")
68          );
69      }

71      @Test
72      @SuppressWarnings("PMD.DoNotUseThreads")
73      public void makesXslTransformationsInThreads() throws Exception {
74          final int loop = 50;
75          final int timeout = 30;
76          final XSL xsl = new XSLDocument(
77              StringUtils.join(
78                  "<xsl:stylesheet  ",
79                  " xmlns:xsl='http://www.w3.org/1999/XSL/Transform' ",
80                  " version='2.0' >",
81                  "<xsl:template match='/'><works/>",
82                  "</xsl:template> </xsl:stylesheet>"
83              )
84          );
85          final Runnable runnable = () -> MatcherAssert.assertThat(
86              xsl.transform(new XMLDocument("<test/>")),
87              XhtmlMatchers.hasXPath("/works")
88          );
89          final ExecutorService service = Executors.newFixedThreadPool(5);
90          for (int count = 0; count < loop; count += 1) {
91              service.submit(runnable);
92          }
93          service.shutdown();
94          MatcherAssert.assertThat(
95              service.awaitTermination(timeout, TimeUnit.SECONDS),
96              Matchers.is(true)
97          );
98          service.shutdownNow();
99      }

149     @Test
150     public void transformsIntoTextWithParams() {
151         final XSL xsl = new XSLDocument(
152             StringUtils.join(
153                 "<xsl:stylesheet   ",
154                 " xmlns:xsl='http://www.w3.org/1999/XSL/Transform'    ",
155                 " xmlns:xs='http://www.w3.org/2001/XMLSchema'",
156                 " version='2.0'><xsl:output method='text'  />",
157                 "<xsl:param name='boom' />",
158                 "<xsl:template match='/'>[<xsl:value-of select='$boom'/>]",
159                 "</xsl:template>   </xsl:stylesheet>"
160             )
161         );
162         MatcherAssert.assertThat(
163             xsl.with("boom", "Donny").applyTo(new XMLDocument("<ehe/>")),
164             Matchers.equalTo("[Donny]")
165         );
166     }

168     @Test
169     public void transformsIntoTextWithIntegerParams() {
170         final XSL xsl = new XSLDocument(
171             StringUtils.join(
172                 "<xsl:stylesheet     ",
173                 " xmlns:xsl='http://www.w3.org/1999/XSL/Transform'       ",
174                 " version='2.0'><xsl:output method='text'    />",
175                 "<xsl:param name='faa' select='5'/>",
176                 "<xsl:template match='/'>+<xsl:value-of select='$faa'/>+",
177                 "</xsl:template>   </xsl:stylesheet>  "
178             )
179         );
180         MatcherAssert.assertThat(
181             xsl.with("faa", 1).applyTo(new XMLDocument("<r0/>")),
182             Matchers.equalTo("+1+")
183         );

186     @Test
187     public void catchesXslErrorMessages() {
188         try {
189             new XSLDocument(
190                 StringUtils.join(
191                     " <xsl:stylesheet",
192                     "  xmlns:xsl='http://www.w3.org/1999/XSL/Transform'",
193                     "  version='2.0'><xsl:template match='/'>",
194                     "<xsl:message terminate='yes'>",
195                     "<xsl:text>oopsie...</xsl:text>",
196                     " </xsl:message></xsl:template></xsl:stylesheet>"
197                 )
198             ).transform(new XMLDocument("<zz1/>"));
199             Assertions.fail("Exception expected here");
200         } catch (final IllegalArgumentException ex) {
201             MatcherAssert.assertThat(
202                 ex.getLocalizedMessage(),
203                 Matchers.containsString("oopsie...")
204             );
205         }
206     }
