52      @Test
53      void makesXslTransformations() {
54          final XSL xsl = new XSLDocument(
55              StringUtils.join(
56                  "<xsl:stylesheet",
57                  " xmlns:xsl='http://www.w3.org/1999/XSL/Transform'",
58                  " version='2.0'>",
59                  "<xsl:template match='/'><done/>",
60                  "</xsl:template></xsl:stylesheet>"
61              )
62          );
63          MatcherAssert.assertThat(
64              xsl.transform(new XMLDocument("<a/>")),
65              XhtmlMatchers.hasXPath("/done")
66          );
67          MatcherAssert.assertThat(
68              xsl.transform(new XMLDocument("<a></a>")),
69              XhtmlMatchers.hasXPath("/done ")
70          );
71      }

73      @Test
74      @SuppressWarnings("PMD.DoNotUseThreads")
75      void makesXslTransformationsInThreads() throws Exception {
76          final int loop = 50;
77          final int timeout = 30;
78          final XSL xsl = new XSLDocument(
79              StringUtils.join(
80                  "<xsl:stylesheet  ",
81                  " xmlns:xsl='http://www.w3.org/1999/XSL/Transform' ",
82                  " version='2.0' >",
83                  "<xsl:template match='/'><works/>",
84                  "</xsl:template> </xsl:stylesheet>"
85              )
86          );
87          final Runnable runnable = () -> MatcherAssert.assertThat(
88              xsl.transform(new XMLDocument("<test/>")),
89              XhtmlMatchers.hasXPath("/works")
90          );
91          final ExecutorService service = Executors.newFixedThreadPool(5);
92          for (int count = 0; count < loop; count += 1) {
93              service.submit(runnable);
94          }
95          service.shutdown();
96          MatcherAssert.assertThat(
97              service.awaitTermination(timeout, TimeUnit.SECONDS),
98              Matchers.is(true)
99          );
100         service.shutdownNow();
101     }

245     @Test
246     void transformsInManyThreads() throws Exception {
247         final XSL xsl = new XSLDocument(
248             this.getClass().getResourceAsStream("first.xsl")
249         ).with(new ClasspathSources());
250         final XML xml = new XMLDocument(
251             this.getClass().getResourceAsStream("simple.xml")
252         );
253         final AtomicInteger done = new AtomicInteger(0);
254         final CountDownLatch latch = new CountDownLatch(1);
255         final int total = 50;
256         final ExecutorService exec = Executors.newFixedThreadPool(total);
257         for (int task = 0; task < total; ++task) {
258             exec.submit(
259                 () -> {
260                     latch.await();
261                     xsl.transform(xml);
262                     xsl.transform(xml);
263                     done.incrementAndGet();
264                     return 0;
265                 }
266             );
267         }
268         latch.countDown();
269         exec.shutdown();
270         exec.awaitTermination(1L, TimeUnit.MINUTES);
271         MatcherAssert.assertThat(done.get(), Matchers.equalTo(total));
272     }
