61      @Test
62      public void findsDocumentNodesWithXpath() {
63          final XML doc = new XMLDocument(
64              "<r><a>\u0443\u0440\u0430!</a><a>B</a></r>"
65          );
66          MatcherAssert.assertThat(
67              doc.xpath("//a/text()"),
68              Matchers.hasSize(2)
69          );
70          MatcherAssert.assertThat(
71              doc.xpath("/r/a/text()"),
72              Matchers.hasItem("\u0443\u0440\u0430!")
73          );
74      }

76      @Test
77      public void findWithXpathListEqualsToJavaUtilList() {
78          MatcherAssert.assertThat(
79              new XMLDocument(
80                  "<does><not><matter/></not></does>"
81              ).xpath("//missing/text()"),
82              new IsEqual<>(
83                  Collections.<String>emptyList()
84              )
85          );
86          MatcherAssert.assertThat(
87              new XMLDocument(
88                  "<root><item>first</item><item>second</item></root>"
89              ).xpath("//root/item[1]/text()"),
90              new IsEqual<>(
91                  Collections.singletonList("first")
92              )
93          );
94          MatcherAssert.assertThat(
95              new XMLDocument(
96                  "<root><item>abc</item><item>def</item></root>"
97              ).xpath("/root/item/text()"),
98              new IsEqual<>(
99                  Arrays.asList("abc", "def")
100             )
101         );
102     }

104     @Test
105     public void findsWithXpathAndNamespaces() {
106         final XML doc = new XMLDocument(
107             // @checkstyle LineLength (1 line)
108             "<html xmlns='http://www.w3.org/1999/xhtml'><div>\u0443\u0440\u0430!</div></html>"
109         );

116             Matchers.hasSize(1)

120     @Test
121     public void findsWithXpathWithCustomNamespace() throws Exception {
122         final File file = Files.createTempDirectory("")
123             .resolve("x.xml").toFile();
124         new LengthOf(
125             new TeeInput(
126                 "<a xmlns='urn:foo'><b>\u0433!</b></a>",
127                 file
128             )
129         ).value();
130         final XML doc = new XMLDocument(file).registerNs("f", "urn:foo");
131         MatcherAssert.assertThat(
132             doc.nodes("/f:a/f:b[.='\u0433!']"),
133             Matchers.hasSize(1)
134         );
135         MatcherAssert.assertThat(
136             doc.xpath("//f:b/text()").get(0),
137             Matchers.equalTo("\u0433!")
138         );
139     }

244     @Test
245     public void parsesInMultipleThreads() throws Exception {
246         final int timeout = 10;
247         final int loop = 100;
248         final Runnable runnable = () -> MatcherAssert.assertThat(
249             new XMLDocument("<root><hey/></root>"),
250             XhtmlMatchers.hasXPath("/root/hey")
251         );
252         final ExecutorService service = Executors.newFixedThreadPool(5);
253         for (int count = 0; count < loop; count = count + 1) {
254             service.submit(runnable);
255         }
256         service.shutdown();
257         MatcherAssert.assertThat(
258             service.awaitTermination(timeout, TimeUnit.SECONDS),
259             Matchers.is(true)
260         );
261         service.shutdownNow();
262     }
